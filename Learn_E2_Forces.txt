@name Learn_E2_Forces
@inputs 
@outputs 
@persist Prop:entity Force
@trigger 

interval(100)

if(first() | duped()) { # stuff goes here
	Force = 100 # How hard to apply the force
	Prop = propSpawn("models/props_combine/breenglobe.mdl",0) # Number at end, 1 for frozen 0 for unfrozen
	Prop:setPos(entity():pos() + vec(0,0,30))
}

if(owner():keyAttack1()) { # Owner of 	the e2 is pressing the mouse 1 button
	Vec = owner():eye()
	Prop:applyForce(Force * Vec * Prop:mass())
}

elseif(owner():keyAttack2()) { # Reset the prop on top of our heads
	Prop:setPos(owner():pos() + vec(0,0,50))
}
